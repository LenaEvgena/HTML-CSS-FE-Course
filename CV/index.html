function getMiddle(s) {
  let length = s.split('').length;
  let arr = s.split('');
  if (length % 2 === 0) {
    return arr.splice(((length / 2) - 1), 2).join('');
  } else {
    return arr.splice(((length / 2) + 0.5) - 1, 1).join('');
  }
}
